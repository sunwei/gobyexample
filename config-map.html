func UnmarshalTo(data []byte, f Format, v any) error {
    var err error

    switch f {
    case TOML:
        err = toml.Unmarshal(data, v)

    default:
        return fmt.Errorf(
            "unmarshal of format %q is not supported", f)
    }

    if err == nil {
        return nil
    }

    return err
}
